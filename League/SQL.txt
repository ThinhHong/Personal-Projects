CREATE TABLE champion(
    id INTEGER PRIMARY KEY,
    champ_name VARCHAR(15) NOT NULL,
    version VARCHAR(10) NOT NULL,

    difficulty INTEGER
);

CREATE TABLE type(
    tags VARCHAR(10),
    id INTEGER,
    PRIMARY KEY (id,tags),
    FOREIGN KEY (id) REFERENCES champion(id) ON DELETE CASCADE
);
